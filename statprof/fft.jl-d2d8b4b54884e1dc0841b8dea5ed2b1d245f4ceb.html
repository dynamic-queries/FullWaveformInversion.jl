113 # FFT plans can be user-created on a different task, whose stream might be different from
114 # the one used in the current task. call this function before every API call that performs
115 # operations on a stream to ensure the plan is using the correct task-local stream.
116 @inline function update_stream(plan::CuFFTPlan)
117 new_stream = stream()
118 if plan.stream != new_stream
119 plan.stream = new_stream
120 cufftSetStream(plan, new_stream)
121
122 # replace the workarea by one (asynchronously) allocated on the current stream
123 new_workarea = similar(plan.workarea)
124 cufftSetWorkArea(plan, new_workarea)
125 CUDA.unsafe_free!(plan.workarea)
126 plan.workarea = new_workarea
127 end
128 return
129 end
