294 # polynomial evaluation using compensated summation.
295 # much more accurate, especially when lo can be combined with other rounding errors
296 @inline function exthorner(x, p::Tuple)
297 hi, lo = p[end], zero(x)
298 for i in length(p)-1:-1:1
299 pi = p[i]
300 prod, err = two_mul(hi,x)
301 hi = pi+prod
302 lo = fma(lo, x, prod - (hi - pi) + err)
303 end
304 return hi, lo
305 end

674 hypot(x::Number, y::Number, xs::Number...) = _hypot(float.(promote(x, y, xs...))...)
675 function _hypot(x, y)
676 # preserves unit
677 axu = abs(x)
678 ayu = abs(y)
679
680 # unitless
681 ax = axu / oneunit(axu)
682 ay = ayu / oneunit(ayu)
683
684 # Return Inf if either or both inputs is Inf (Compliance with IEEE754)
685 if isinf(ax) || isinf(ay)
686 return typeof(axu)(Inf)
687 end
688
689 # Order the operands
690 if ay > ax
691 axu, ayu = ayu, axu
692 ax, ay = ay, ax
693 end
694
695 # Widely varying operands
696 if ay <= ax*sqrt(eps(typeof(ax))/2) #Note: This also gets ay == 0
697 return axu
698 end
699
700 # Operands do not vary widely
701 scale = eps(typeof(ax))*sqrt(floatmin(ax)) #Rescaling constant
702 if ax > sqrt(floatmax(ax)/2)
703 ax = ax*scale
704 ay = ay*scale
705 scale = inv(scale)
706 elseif ay < sqrt(floatmin(ax))
707 ax = ax/scale
708 ay = ay/scale
709 else
710 scale = oneunit(scale)
711 end
712 h = sqrt(muladd(ax, ax, ay*ay))
713 # This branch is correctly rounded but requires a native hardware fma.
714 if Core.Intrinsics.have_fma(typeof(h))
715 hsquared = h*h
716 axsquared = ax*ax
717 h -= (fma(-ay, ay, hsquared-axsquared) + fma(h, h,-hsquared) - fma(ax, ax, -axsquared))/(2*h)
718 # This branch is within one ulp of correctly rounded.
719 else
720 if h <= 2*ay
721 delta = h-ay
722 h -= muladd(delta, delta-2*(ax-ay), ax*(2*delta - ax))/(2*h)
723 else
724 delta = h-ax
725 h -= muladd(delta, delta, muladd(ay, (4*delta - ay), 2*delta*(ax - 2*ay)))/(2*h)
726 end
727 end
728 return h*scale*oneunit(axu)
729 end

772 """
773 ldexp(x, n)
774
775 Compute ``x \\times 2^n``.
776
777 # Examples
778 ```jldoctest
779 julia> ldexp(5., 2)
780 20.0
781 ```
782 """
783 function ldexp(x::T, e::Integer) where T<:IEEEFloat
784 xu = reinterpret(Unsigned, x)
785 xs = xu & ~sign_mask(T)
786 xs >= exponent_mask(T) && return x # NaN or Inf
787 k = (xs >> significand_bits(T)) % Int
788 if k == 0 # x is subnormal
789 xs == 0 && return x # +-0
790 m = leading_zeros(xs) - exponent_bits(T)
791 ys = xs << unsigned(m)
792 xu = ys | (xu & sign_mask(T))
793 k = 1 - m
794 # underflow, otherwise may have integer underflow in the following n + k
795 e < -50000 && return flipsign(T(0.0), x)
796 end
797 # For cases where e of an Integer larger than Int make sure we properly
798 # overflow/underflow; this is optimized away otherwise.
799 if e > typemax(Int)
800 return flipsign(T(Inf), x)
801 elseif e < typemin(Int)
802 return flipsign(T(0.0), x)
803 end
804 n = e % Int
805 k += n
806 # overflow, if k is larger than maximum possible exponent
807 if k >= exponent_raw_max(T)
808 return flipsign(T(Inf), x)
809 end
810 if k > 0 # normal case
811 xu = (xu & ~exponent_mask(T)) | (rem(k, uinttype(T)) << significand_bits(T))
812 return reinterpret(T, xu)
813 else # subnormal case
814 if k <= -significand_bits(T) # underflow
815 # overflow, for the case of integer overflow in n + k
816 e > 50000 && return flipsign(T(Inf), x)
817 return flipsign(T(0.0), x)
818 end
819 k += significand_bits(T)
820 # z = T(2.0) ^ (-significand_bits(T))
821 z = reinterpret(T, rem(exponent_bias(T)-significand_bits(T), uinttype(T)) << significand_bits(T))
822 xu = (xu & ~exponent_mask(T)) | (rem(k, uinttype(T)) << significand_bits(T))
823 return z*reinterpret(T, xu)
824 end
825 end

874

876 a non-zero finite number, then the result will be a number of the same type and
877 sign as `x`, and whose absolute value is on the interval ``[1,2)``. Otherwise
878 `x` is returned.

892 (-Inf, Inf, NaN)

895 function significand(x::T) where T<:IEEEFloat
896 xu = reinterpret(Unsigned, x)

898 xs >= exponent_mask(T) && return x # NaN or Inf
899 if xs <= (~exponent_mask(T) & ~sign_mask(T)) # x is subnormal
900 xs == 0 && return x # +-0
901 m = unsigned(leading_zeros(xs) - exponent_bits(T))
902 xs <<= m
903 xu = xs | (xu & sign_mask(T))
904 end

946 function rem(x::T, p::T, ::RoundingMode{:Nearest}) where T<:IEEEFloat
947 (iszero(p) || !isfinite(x) || isnan(p)) && return T(NaN)
948 x == p && return copysign(zero(T), x)
949 oldx = x
950 x = abs(rem(x, 2p)) # 2p may overflow but that's okay
951 p = abs(p)
952 if p < 2 * floatmin(T) # Check whether dividing p by 2 will underflow
953 if 2x > p
954 x -= p
955 if 2x >= p
956 x -= p
957 end
958 end
959 else
960 p_half = p / 2
961 if x > p_half
962 x -= p
963 if x >= p_half
964 x -= p
965 end
966 end
967 end
968 return flipsign(x, oldx)

1086 function add22condh(xh::Float64, xl::Float64, yh::Float64, yl::Float64)
1087 # This algorithm, due to Dekker, computes the sum of two
1088 # double-double numbers and returns the high double. References:
1089 # [1] http://www.digizeitschriften.de/en/dms/img/?PID=GDZPPN001170007
1090 # [2] https://doi.org/10.1007/BF01397083
1091 r = xh+yh
1092 s = (abs(xh) > abs(yh)) ? (xh-r+yh+yl+xl) : (yh-r+xh+xl+yl)
1093 zh = r+s
1094 return zh
1095 end

1098 const pi1o2_h = 1.5707963267948966 # convert(Float64, pi * BigFloat(1/2))

1100
1101 const pi2o2_h = 3.141592653589793 # convert(Float64, pi * BigFloat(1))
1102 const pi2o2_l = 1.2246467991473532e-16 # convert(Float64, pi * BigFloat(1) - pi2o2_h)
1103
1104 const pi3o2_h = 4.71238898038469 # convert(Float64, pi * BigFloat(3/2))
1105 const pi3o2_l = 1.8369701987210297e-16 # convert(Float64, pi * BigFloat(3/2) - pi3o2_h)
1106
1107 const pi4o2_h = 6.283185307179586 # convert(Float64, pi * BigFloat(2))

1110 """
1111 rem2pi(x, r::RoundingMode)
1112
1113 Compute the remainder of `x` after integer division by `2π`, with the quotient rounded
1114 according to the rounding mode `r`. In other words, the quantity
1115
1116 x - 2π*round(x/(2π),r)
1117
1118 without any intermediate rounding. This internally uses a high precision approximation of
1119 2π, and so will give a more accurate result than `rem(x,2π,r)`
1120
1121 - if `r == RoundNearest`, then the result is in the interval ``[-π, π]``. This will generally
1122 be the most accurate result. See also [`RoundNearest`](@ref).
1123
1124 - if `r == RoundToZero`, then the result is in the interval ``[0, 2π]`` if `x` is positive,.
1125 or ``[-2π, 0]`` otherwise. See also [`RoundToZero`](@ref).
1126
1127 - if `r == RoundDown`, then the result is in the interval ``[0, 2π]``.
1128 See also [`RoundDown`](@ref).
1129 - if `r == RoundUp`, then the result is in the interval ``[-2π, 0]``.
1130 See also [`RoundUp`](@ref).
1131
1132 # Examples
1133 ```jldoctest
1134 julia> rem2pi(7pi/4, RoundNearest)
1135 -0.7853981633974485
1136
1137 julia> rem2pi(7pi/4, RoundDown)
1138 5.497787143782138
1139 ```
1140 """
1141 function rem2pi end
1142 function rem2pi(x::Float64, ::RoundingMode{:Nearest})
1143 isfinite(x) || return NaN
1144
1145 abs(x) < pi && return x
1146
1147 n,y = rem_pio2_kernel(x)
1148
1149 if iseven(n)
1150 if n & 2 == 2 # n % 4 == 2: add/subtract pi
1151 if y.hi <= 0
1152 return add22condh(y.hi,y.lo,pi2o2_h,pi2o2_l)
1153 else
1154 return add22condh(y.hi,y.lo,-pi2o2_h,-pi2o2_l)
1155 end
1156 else # n % 4 == 0: add 0
1157 return y.hi+y.lo
1158 end
1159 else
1160 if n & 2 == 2 # n % 4 == 3: subtract pi/2
1161 return add22condh(y.hi,y.lo,-pi1o2_h,-pi1o2_l)
1162 else # n % 4 == 1: add pi/2
1163 return add22condh(y.hi,y.lo,pi1o2_h,pi1o2_l)
1164 end
1165 end
