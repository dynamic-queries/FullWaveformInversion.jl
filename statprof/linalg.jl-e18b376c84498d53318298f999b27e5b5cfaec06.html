409 function LinearAlgebra.norm(v::AbstractGPUArray{T}, p::Real=2) where {T}
410 result_type, sum_type, promote_ = _normtypes(T)
411 isempty(v) && return zero(result_type)
412 p == 0 && return convert(result_type, count(!iszero, v))
413 spp = convert(sum_type, p)
414 init = zero(sum_type) # To set the accumulation type in `sum`
415 # Rescaling heuristic similar to Base, see LinearAlgebra/src/generic.jl
416 result = if p > 1 || p < -1 # May need rescaling
417 infnorm = p > 1 ? maximum(norm, v) : minimum(norm, v)
418 if isinf(p) || iszero(infnorm) || isinf(infnorm)
419 return convert(result_type, infnorm) # Return early to skip conversions
420 end
421 factor = convert(sum_type, infnorm)
422 if p == 2
423 if isfinite(length(v) * factor^2) && !iszero(factor^2) # No rescaling
424 sqrt(sum(x -> LinearAlgebra.norm_sqr(promote_(x)), v; init=init))
425 else # Rescaling
426 factor * sqrt(sum(x -> (norm(promote_(x)) / factor)^2, v; init=init))
427 end
428 else
429 if isfinite(length(v) * factor^spp) && !iszero(factor^spp) # No rescaling
430 sum(x -> norm(promote_(x))^spp, v; init=init)^inv(spp)
431 else # Rescaling
432 factor * (sum(x -> (norm(promote_(x)) / factor)^spp, v; init=init)^inv(spp))
433 end
434 end
435 elseif p == 1
436 sum(x -> norm(promote_(x)), v; init=init)
437 else
438 sum(x -> norm(promote_(x))^spp, v; init=init)^inv(spp)
439 end
440 return convert(result_type, result)
441 end

443 function _normtypes(::Type{T}) where {T}
444 result_type = typeof(float(norm(zero(T))))
445 # Accumulate in at least Float32, like nrm2 in CUBLAS
446 sum_type = promote_type(Float32, result_type)
447 # If sum_type is wider than T, promote before applying other functions. To work in GPU
448 # kernels this operation must close around a value, not a type, hence the prototype
449 prototype = zero(promote_type(T, sum_type))
450 promote_(x) = convert(typeof(prototype), x)
451 return result_type, sum_type, promote_
452 end

518 Base.iszero(x::AbstractGPUMatrix{T}) where {T} = all(iszero, x)
519 function Base.isone(x::AbstractGPUMatrix{T}) where {T}
520 n,m = size(x)
521 m != n && return false
522
523 # lazily perform `x-I`
524 bc = Broadcast.broadcasted(x, CartesianIndices(x)) do _x, inds
525 _x - (inds[1] == inds[2] ? one(_x) : zero(_x))
526 end
527 # call `GPUArrays.mapreducedim!` directly, which supports Broadcasted inputs
528 y = similar(x, Bool, 1)
529 GPUArrays.mapreducedim!(iszero, &, y, Broadcast.instantiate(bc); init=true)
530
531 Array(y)[]
532 end
