3 # Twice-precision arithmetic.
4
5 # Necessary for creating nicely-behaved ranges like r = 0.1:0.1:0.3
6 # that return r[3] == 0.3. Otherwise, we have roundoff error due to
7 # 0.1 + 2*0.1 = 0.30000000000000004
8
9 """
10 hi, lo = splitprec(F::Type{<:AbstractFloat}, i::Integer)
11
12 Represent an integer `i` as a pair of floating-point numbers `hi` and
13 `lo` (of type `F`) such that:
14 - `widen(hi) + widen(lo) ≈ i`. It is exact if 1.5 * (number of precision bits in `F`) is greater than the number of bits in `i`.
15 - all bits in `hi` are more significant than any of the bits in `lo`
16 - `hi` can be exactly multiplied by the `hi` component of another call to `splitprec`.
17
18 In particular, while `convert(Float64, i)` can be lossy since Float64
19 has only 53 bits of precision, `splitprec(Float64, i)` is exact for
20 any Int64/UInt64.
21 """
22 function splitprec(::Type{F}, i::Integer) where {F<:AbstractFloat}
23 hi = truncbits(F(i), cld(precision(F), 2))
24 ihi = oftype(i, hi)
25 hi, F(i - ihi)
26 end

42 hi, lo = canonicalize2(big, little)

44 Generate a representation where all the nonzero bits in `hi` are more
45 significant than any of the nonzero bits in `lo`. `big` must be larger

48 function canonicalize2(big, little)
49 h = big+little

53 """
54 zhi, zlo = add12(x, y)
55
56 A high-precision representation of `x + y` for floating-point
57 numbers. Mathematically, `zhi + zlo = x + y`, where `zhi` contains the
58 most significant bits and `zlo` the least significant.
59
60 Because of the way floating-point numbers are printed, `lo` may not
61 look the way you might expect from the standpoint of decimal
62 representation, even though it is exact from the standpoint of binary
63 representation.
64
65 Example:
66 ```julia-repl
67 julia> 1.0 + 1.0001e-15
68 1.000000000000001
69
70 julia> big(1.0) + big(1.0001e-15)
71 1.000000000000001000100000000000020165767380775934141445417482375879192346701529
72
73 julia> hi, lo = Base.add12(1.0, 1.0001e-15)
74 (1.000000000000001, -1.1012302462515652e-16)
75
76 julia> big(hi) + big(lo)
77 1.000000000000001000100000000000020165767380775934141445417482375879192346701529
78 ```
79
80 `lo` differs from 1.0e-19 because `hi` is not exactly equal to
81 the first 16 decimal digits of the answer.
82 """
83 function add12(x::T, y::T) where {T}
84 x, y = ifelse(abs(y) > abs(x), (y, x), (x, y))
85 canonicalize2(x, y)
86 end
87 add12(x, y) = add12(promote(x, y)...)

89 """
90 zhi, zlo = mul12(x, y)
91
92 A high-precision representation of `x * y` for floating-point
93 numbers. Mathematically, `zhi + zlo = x * y`, where `zhi` contains the
94 most significant bits and `zlo` the least significant.
95
96 Example:
97 ```julia-repl
98 julia> x = Float32(π)
99 3.1415927f0
100
101 julia> x * x
102 9.869605f0
103
104 julia> Float64(x) * Float64(x)
105 9.869604950382893
106
107 julia> hi, lo = Base.mul12(x, x)
108 (9.869605f0, -1.140092f-7)
109
110 julia> Float64(hi) + Float64(lo)
111 9.869604950382893
112 ```
113 """
114 function mul12(x::T, y::T) where {T<:AbstractFloat}
115 h = x * y
116 ifelse(iszero(h) | !isfinite(h), (h, h), canonicalize2(h, fma(x, y, -h)))
117 end
118 mul12(x::T, y::T) where {T} = (p = x * y; (p, zero(p)))
119 mul12(x, y) = mul12(promote(x, y)...)

121 """
122 zhi, zlo = div12(x, y)
123
124 A high-precision representation of `x / y` for floating-point
125 numbers. Mathematically, `zhi + zlo ≈ x / y`, where `zhi` contains the
126 most significant bits and `zlo` the least significant.
127
128 Example:
129 ```julia-repl
130 julia> x, y = Float32(π), 3.1f0
131 (3.1415927f0, 3.1f0)
132
133 julia> x / y
134 1.013417f0
135
136 julia> Float64(x) / Float64(y)
137 1.0134170444063078
138
139 julia> hi, lo = Base.div12(x, y)
140 (1.013417f0, 3.8867366f-8)
141
142 julia> Float64(hi) + Float64(lo)
143 1.0134170444063066
144 """
145 function div12(x::T, y::T) where {T<:AbstractFloat}
146 # We lose precision if any intermediate calculation results in a subnormal.
147 # To prevent this from happening, standardize the values.
148 xs, xe = frexp(x)
149 ys, ye = frexp(y)
150 r = xs / ys
151 rh, rl = canonicalize2(r, -fma(r, ys, -xs)/ys)
152 ifelse(iszero(r) | !isfinite(r), (r, r), (ldexp(rh, xe-ye), ldexp(rl, xe-ye)))
153 end
154 div12(x::T, y::T) where {T} = (p = x / y; (p, zero(p)))
155 div12(x, y) = div12(promote(x, y)...)

158 ## TwicePrecision

160 """
161 TwicePrecision{T}(hi::T, lo::T)
162 TwicePrecision{T}((num, denom))
163
164 A number with twice the precision of `T`, e.g., quad-precision if `T =
165 Float64`.
166
167 !!! warn
168 `TwicePrecision` is an internal type used to increase the
169 precision of floating-point ranges, and not intended for external use.
170 If you encounter them in real code, the most likely explanation is
171 that you are directly accessing the fields of a range. Use
172 the function interface instead, `step(r)` rather than `r.step`
173
174 # Extended help
175
176 `hi` represents the high bits (most significant bits) and
177 `lo` the low bits (least significant bits). Rational values
178 `num//denom` can be approximated conveniently using the syntax
179 `TwicePrecision{T}((num, denom))`.
180
181 When used with `T<:Union{Float16,Float32,Float64}` to construct an "exact"
182 `StepRangeLen`, `ref` should be the range element with smallest
183 magnitude and `offset` set to the corresponding index. For
184 efficiency, multiplication of `step` by the index is not performed at
185 twice precision: `step.hi` should have enough trailing zeros in its
186 `bits` representation that `(0:len-1)*step.hi` is exact (has no
187 roundoff error). If `step` has an exact rational representation
188 `num//denom`, then you can construct `step` using
189
190 step = TwicePrecision{T}((num, denom), nb)
191
192 where `nb` is the number of trailing zero bits of `step.hi`. For
193 ranges, you can set `nb = ceil(Int, log2(len-1))`.
194 """

411 function (:)(start::T, step::T, stop::T) where T<:IEEEFloat

414 # perform all computations in terms of the rationals)

417 start_n, start_d = rat(start)

419 if start_d != 0 && stop_d != 0 &&

421 den = lcm_unchecked(start_d, step_d) # use same denominator for start and step
422 m = maxintfloat(T, Int)
423 if den != 0 && abs(start*den) <= m && abs(step*den) <= m && # will round succeed?
424 rem(den, start_d) == 0 && rem(den, step_d) == 0 # check lcm overflow

427 len = max(0, Int(div(den*stop_n - stop_d*start_n + step_n*stop_d, step_n*stop_d)))
428 # Integer ops could overflow, so check that this makes sense
429 if isbetween(start, start + (len-1)*step, stop + step/2) &&
430 !isbetween(start, start + len*step, stop)
431 # Return a 2x precision range
432 return floatrange(T, start_n, step_n, len, den)
433 end
434 end
435 end
436 end
437 # Fallback, taking start and step literally
438 # n.b. we use Int as the default length type for IEEEFloats
439 lf = (stop-start)/step
440 if lf < 0
441 len = 0
442 elseif lf == 0

581 function _convertSRL(::Type{StepRangeLen{T,R,S,L}}, r::AbstractRange{U}) where {T,R,S,L,U}
582 # if start and step have a rational approximation in the old type,
583 # then we transfer that rational approximation to the new type
584 f, s = first(r), step(r)
585 start_n, start_d = rat(f)
586 step_n, step_d = rat(s)
587 if start_d != 0 && step_d != 0 &&
588 U(start_n/start_d) == f && U(step_n/step_d) == s
589 den = lcm_unchecked(start_d, step_d)
590 m = maxintfloat(T, Int)
591 if den != 0 && abs(f*den) <= m && abs(s*den) <= m &&
592 rem(den, start_d) == 0 && rem(den, step_d) == 0
593 start_n = round(Int, f*den)
594 step_n = round(Int, s*den)
595 return floatrange(T, start_n, step_n, L(length(r)), den)
596 end
597 end
598 return __convertSRL(StepRangeLen{T,R,S,L}, r)
599 end

608 function sum(r::StepRangeLen)
609 l = length(r)
610 # Compute the contribution of step over all indices.
611 # Indexes on opposite side of r.offset contribute with opposite sign,
612 # r.step * (sum(1:np) - sum(1:nn))
613 np, nn = l - r.offset, r.offset - 1 # positive, negative
614 # To prevent overflow in sum(1:n), multiply its factors by the step
615 sp, sn = sumpair(np), sumpair(nn)
616 W = widen(typeof(l))
617 Δn = W(sp[1]) * W(sp[2]) - W(sn[1]) * W(sn[2])
618 s = r.step * Δn
619 # Add in contributions of ref
620 ref = r.ref * l
621 convert(eltype(r), s + ref)
622 end
623 function sum(r::StepRangeLen{<:Any,<:TwicePrecision,<:TwicePrecision})
624 l = length(r)
625 # Compute the contribution of step over all indices.
626 # Indexes on opposite side of r.offset contribute with opposite sign,
627 # r.step * (sum(1:np) - sum(1:nn))
628 np, nn = l - r.offset, r.offset - 1 # positive, negative
629 # To prevent overflow in sum(1:n), multiply its factors by the step
630 sp, sn = sumpair(np), sumpair(nn)
631 tp = _tp_prod(r.step, sp[1], sp[2])
632 tn = _tp_prod(r.step, sn[1], sn[2])
633 s_hi, s_lo = add12(tp.hi, -tn.hi)
634 s_lo += tp.lo - tn.lo
635 # Add in contributions of ref
636 ref = r.ref * l
637 sm_hi, sm_lo = add12(s_hi, ref.hi)
638 add12(sm_hi, sm_lo + ref.lo)[1]
639 end
640
641 # sum(1:n) as a product of two integers
642 sumpair(n::Integer) = iseven(n) ? (n+1, n>>1) : (n, (n+1)>>1)

644 function +(r1::StepRangeLen{T,R}, r2::StepRangeLen{T,R}) where T where R<:TwicePrecision
645 len = length(r1)
646 (len == length(r2) ||
647 throw(DimensionMismatch("argument dimensions must match")))
648 if r1.offset == r2.offset
649 imid = r1.offset
650 ref = r1.ref + r2.ref
651 else
652 imid = round(typeof(len), (r1.offset+r2.offset)/2)
653 ref1mid = _getindex_hiprec(r1, imid)
654 ref2mid = _getindex_hiprec(r2, imid)
655 ref = ref1mid + ref2mid
656 end
657 step = twiceprecision(r1.step + r2.step, nbitslen(T, len, imid))
658 StepRangeLen{T,typeof(ref),typeof(step),typeof(len)}(ref, step, len, imid)
659 end

663 # For Float16, Float32, and Float64, this returns a StepRangeLen
664 function range_start_stop_length(start::T, stop::T, len::Integer) where {T<:IEEEFloat}
665 len = len + 0 # promote with Int
666 len < 2 && return _linspace1(T, start, stop, len)
667 if start == stop
668 return steprangelen_hp(T, start, zero(T), 0, len, 1)
669 end
670 # Attempt to find exact rational approximations
671 start_n, start_d = rat(start)
672 stop_n, stop_d = rat(stop)
673 if start_d != 0 && stop_d != 0
674 den = lcm_unchecked(start_d, stop_d)
675 m = maxintfloat(T, Int)
676 if den != 0 && abs(den*start) <= m && abs(den*stop) <= m
677 start_n = round(Int, den*start)
678 stop_n = round(Int, den*stop)
679 if T(start_n/den) == start && T(stop_n/den) == stop
680 return _linspace(T, start_n, stop_n, len, den)
681 end
682 end
683 end
684 _linspace(start, stop, len)
685 end
686
687 function _linspace(start::T, stop::T, len::Integer) where {T<:IEEEFloat}
688 len = len + 0 # promote with Int
689 (isfinite(start) && isfinite(stop)) || throw(ArgumentError("start and stop must be finite, got $start and $stop"))
690 # Find the index that returns the smallest-magnitude element
691 Δ, Δfac = stop-start, 1
692 if !isfinite(Δ) # handle overflow for large endpoints
693 Δ, Δfac = stop/len - start/len, len
694 end
695 tmin = -(start/Δ)/Δfac # t such that (1-t)*start + t*stop == 0
696 L = typeof(len)
697 lenn1 = len - oneunit(L)
698 imin = round(L, tmin*lenn1 + 1) # index approximately corresponding to t
699 if 1 < imin < len
700 # The smallest-magnitude element is in the interior
701 t = (imin - 1)/lenn1
702 ref = T((1-t)*start + t*stop)
703 step = imin-1 < len-imin ? (ref-start)/(imin-1) : (stop-ref)/(len-imin)
704 elseif imin <= 1
705 imin = oneunit(L)
706 ref = start
707 step = (Δ/(lenn1))*Δfac
708 else
709 imin = len
710 ref = stop
711 step = (Δ/(lenn1))*Δfac
712 end
713 if len == 2 && !isfinite(step)
714 # For very large endpoints where step overflows, exploit the
715 # split-representation to handle the overflow
716 return steprangelen_hp(T, start, (-start, stop), 0, len, oneunit(L))
717 end
718 # 2x calculations to get high precision endpoint matching while also
719 # preventing overflow in ref_hi+(i-offset)*step_hi
720 m, k = prevfloat(floatmax(T)), max(imin-1, len-imin)
721 step_hi_pre = clamp(step, max(-(m+ref)/k, (-m+ref)/k), min((m-ref)/k, (m+ref)/k))
722 nb = nbitslen(T, len, imin)
723 step_hi = truncbits(step_hi_pre, nb)
724 x1_hi, x1_lo = add12((1-imin)*step_hi, ref)
725 x2_hi, x2_lo = add12((len-imin)*step_hi, ref)
726 a, b = (start - x1_hi) - x1_lo, (stop - x2_hi) - x2_lo
727 step_lo = (b - a)/(len - 1)
728 ref_lo = a - (1 - imin)*step_lo
729 steprangelen_hp(T, (ref, ref_lo), (step_hi, step_lo), 0, len, imin)
730 end

732 # range for rational numbers, start = start_n/den, stop = stop_n/den
733 # Note this returns a StepRangeLen
734 _linspace(::Type{T}, start::Integer, stop::Integer, len::Integer) where {T<:IEEEFloat} = _linspace(T, start, stop, len, one(start))
735 function _linspace(::Type{T}, start_n::Integer, stop_n::Integer, len::Integer, den::Integer) where T<:IEEEFloat
736 len = len + 0 # promote with Int
737 len < 2 && return _linspace1(T, start_n/den, stop_n/den, len)
738 L = typeof(len)
739 start_n == stop_n && return steprangelen_hp(T, (start_n, den), (zero(start_n), den), 0, len, oneunit(L))
740 tmin = -start_n/(Float64(stop_n) - Float64(start_n))
741 imin = round(typeof(len), tmin*(len-1)+1)
742 imin = clamp(imin, oneunit(L), len)
743 W = widen(L)
744 start_n = W(start_n)
745 stop_n = W(stop_n)
746 ref_num = W(len-imin) * start_n + W(imin-1) * stop_n
747 ref_denom = W(len-1) * den
748 ref = (ref_num, ref_denom)
749 step_full = (stop_n - start_n, ref_denom)
750 steprangelen_hp(T, ref, step_full, nbitslen(T, len, imin), len, imin)
751 end

753 # For len < 2
754 function _linspace1(::Type{T}, start, stop, len::Integer) where T<:IEEEFloat
755 len >= 0 || throw(ArgumentError("range($start, stop=$stop, length=$len): negative length"))
756 if len <= 1
757 len == 1 && (start == stop || throw(ArgumentError("range($start, stop=$stop, length=$len): endpoints differ")))
758 # Ensure that first(r)==start and last(r)==stop even for len==0
759 # The output type must be consistent with steprangelen_hp

761 return StepRangeLen{T}(Float64(start), Float64(start) - Float64(stop), len, 1)
762 else # T == Float64
763 return StepRangeLen(TwicePrecision(start, zero(T)), TwicePrecision(start, -stop), len, 1)
764 end
765 end
766 throw(ArgumentError("should only be called for len < 2, got $len"))
767 end

771 # Approximate x with a rational representation as a pair of Int values.
772 # Guaranteed to return, but not guaranteed to return a precise answer.
773 # https://en.wikipedia.org/wiki/Continued_fraction#Best_rational_approximations
774 function rat(x)
775 y = x
776 a = d = 1
777 b = c = 0
778 m = maxintfloat(narrow(typeof(x)), Int)
779 while abs(y) <= m
780 f = trunc(Int, y)
781 y -= f
782 a, c = f*a + c, a
783 b, d = f*b + d, b
784 max(abs(a), abs(b)) <= convert(Int,m) || return c, d
785 oftype(x,a)/oftype(x,b) == x && break
786 y = inv(y)
787 end
788 return a, b
789 end
